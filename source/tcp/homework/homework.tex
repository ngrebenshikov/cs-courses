\section{Работа по модулю ``Теоретические модели вычислительных процессов''}

\subsection{Задания}

\begin{enumerate}
	\item Найти $\traces(P)$ (см. Свой вариант).
	\item Найти $\traces(P)$ (см. Свой вариант).
	\item Найти (см. Свой вариант)
		\begin{enumerate}
			\item $traces(P \parallel Q)$
			\item $traces(P \extchoice Q)$
			\item $traces(P \interleave Q)$
		\end{enumerate}
	\item Проверить истинность выражений(см. Свой вариант).
	\item
	\textbf{Дано:}  Представьте супермаркет и покупателя в нем. Покупатель задан описанным ниже процессом ПОК. Событие "положить" означает положить товар с полки в корзину. Событие "достать" означает достать товар из корзины и положить на полку. 
ПОК= положить $\then$ ПОК $\extchoice$  достать $\then$ ПОК $\extchoice$  заплатить $\then$ ПОК $\extchoice$ уйти $\then$ ПОК

\textbf{Найти:}  Постройте новый процесс на основе взаимодействия процесса ПОК и других новых процессов, так чтобы получившийся удовлетворял следующим требованиям:
\begin{enumerate}
	\item Достать из корзины можно только, если там что-то есть.
	\item Корзина максимум вмещает 10 вещей.
	\item Уйти можно только, если корзина пуста или после оплаты.
\end{enumerate}
	
\end{enumerate}

\subsection{Вариант 1.}
\begin{enumerate}
\item $\alpha P=\{a,b,c,d,e\}, P=a \then b \then c \then \STOP$
\item $\alpha P=\{a,b,c,d,e\}, P=a \then (b \then c \then STOP) \barchoice (a \then c \then STOP)$
\item $\alpha P=\{a,b,c,d\},P= a \then b \then c \then \STOP, \newline\alpha Q= \{a,b,d,f\}, Q = a \then b \then d \then \STOP$
\item 
	\textbf{Дано:} 
	
	$P = \mu X \spot a \then b \then c \then X,\newline
	Q = \mu X \spot a \then b \then c \then X \extchoice \mu X \spot b \then c \then a \then X, \newline
	S=(tr \cnt a - tr \cnt c \leq 0)$

\textbf{Найти: }
\begin{enumerate}
	\item верно ли выражение $(P / \trace{a,b}) \sat S$
	\item верно ли выражение $(Q / \trace{a,b}) \sat S$
\end{enumerate}

\end{enumerate}

\subsection{Вариант 2.}

	S=(tr \cnt a - tr \cnt c \leq 0)$

\textbf{Найти: }

	\item верно ли выражение $(P / \trace{a,b,c}) \sat S$
	\item верно ли выражение $(Q / \trace{b,c}) \sat S$

\end{enumerate}
