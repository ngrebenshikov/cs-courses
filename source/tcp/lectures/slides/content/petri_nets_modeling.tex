% ----------------------------------------------------------------
\begin{slide}
	\textbf{События и условия}
	
	\textit{Событие} - дейстие, имеющее место в системе.
	
	\textit{Состояние системы} - множество условий.
	
	\textit{Условие} - предикат или логическое описание состояния системы.
	
	\textit{Предусловия события} - условия необходимые для того, чтобы событие могло произойти.
	
	\textit{Постусловия} - условия, к которым приводит событие.
\end{slide}

% ----------------------------------------------------------------
\begin{slide}
	\textbf{Пример. Автомат-продавец}

	\textit{Условия:} \textbf{а)} ждет, \textbf{б)} заказ прибыл и ждет, \textbf{в)} автомат выполняет заказ, \textbf{г)} заказ выполнен.
	
	\textit{События:} \textbf{1.} Заказ поступил. \textbf{2.} Автомат начинает выполнять заказ. \textbf{3.} Автомат заканчивает выполнение заказа. \textbf{4.} Заказ доставляется.

\begin{slide}
	\textbf{Особенности сетей Петри}
	
\begin{itemize}
	\item параллелизм;
	\item асинхронность (отсутствует течение (измерение) времени);
	\item недетерминированность;
	\item мгновенность и неодновременность событий (примитивность событий);
\end{itemize}
\end{slide}

% ----------------------------------------------------------------
