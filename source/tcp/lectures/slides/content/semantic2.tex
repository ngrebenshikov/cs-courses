\begin{slide}
	\textbf{Аксиоматическое определение оператора цикла}
	
	\textbf{do} $BB \rightarrow$ \textbf{if} $B_1 \Rightarrow S_1 \ ;\ B_2 \Rightarrow S_2 \ ;\ \ldots \ ;\ B_n \Rightarrow S_n$ \textbf{fi} \textbf{od}, где $BB$ - дизъюнкция охран
	
\end{slide}

% ----------------------------------------------------------------
\begin{slide}
	\textbf{Алгоритм Евклида}
	
	задать (N, M);

	\textbf{if} N > 0 $\wedge$ M > 0 $\rightarrow$ n, m := N, M;

	\textbf{do} (n > m) $\rightarrow$ n := n – m ; (m > n) $\rightarrow$ m := m – n \textbf{od};

	выдать (n)

\textbf{fi}.
\end{slide}

\begin{slide}
	\textbf{Аксиоматическое определение оператора цикла}
	
	$H_0(R) = \neg BB \wedge R$ - цикл завершается за 0 шагов.
	
	$H_k(R) = wp(IF, H_{k-1}(R)), k > 0$
	
	$wp(DO, R) = (\exists k: k \geq 0: H_k(R))$
\end{slide}

% ----------------------------------------------------------------
\begin{slide}
	\textbf{Пример цикла}
	
	$i,s := 1, b[0]$
	
	do $i < 11 \rightarrow i,s := i+1, s+b[i]$ od
	
	$\{R: s = (\sum{k:0 \leq k \leq 11: b[k]})\}$

	Инвариант цикла - $\{P: 1 \leq i \leq 11 \wedge s = (\sum{k:0 \leq k \leq i: b[k]})\}$

\end{slide}

% ----------------------------------------------------------------
\begin{slide}
	\textbf{Найдите инвариант цикла}
	
	$\{b \geq 0\}$
	
	$x,y,z := a, b, 0$
	
	do 
	
	$y > 0 \wedge even(y) \rightarrow y,x := y/2, x+x$ 
	
	$odd(y) \rightarrow y,z := y-1, z+x$ 
	
	od
	
	$\{R: z = a \cdot b\}$
\end{slide}

% ----------------------------------------------------------------
\begin{slide}
	\textbf{Ответ:}	$\{P : y \geq 0 \wedge z + x \cdot y = a \cdot b\}$	
	
	\textbf{Доказательство}
	
	Предусловие: $wp((x,y,z := a, b, 0), P) = b \geq 0 \wedge 0 + a \cdot b = a \cdot b$

	$y$ - четно: $wp((y,x := y/2, x+x), P) = y \geq 0 \wedge z + x/2 \cdot (y + y) = y \geq 0 \wedge z + x \cdot y$

	$y$ - нечетно: $wp((y,z := y-1, z+x), P) = y \geq 0 \wedge z - x + x \cdot (y + 1) = y \geq 0 \wedge z + x \cdot y$
	
	Постусловие:	$P \wedge \neg (y > 0 \wedge even(y)) \wedge \neg odd(y) = y = 0 \wedge z + x \cdot y = a \cdot b \Rightarrow z = a \cdot b$
\end{slide}

\begin{slide}
	\textbf{Интерпретация аксиоматической семантики}
	
	Система - набор аксиом в рамках некоторой формальной логической системы, в которой есть правила вывода и/или интерпретации определяемых объектов.
	
	Аксиоматическое описание - $\left\langle S, A\right\rangle$, где $S$ - сигнатура используемых в системе символов функций и констант , а $A$ - набор аксиом.
\end{slide}

	Мощный инструмент для исследования доказательств правильности программ и анализа программ.
	
	Ограниченное использование для описания содержания языков программирования.
	
	\textit{Определение аксиом и правил логического вывода для некоторых операторов языков программирования - очень сложная задача.}

\begin{slide}
	\textbf{Семантическая функция(отображение)}
	
	$M_b('0')=0, M_b('1')=1;$
	
	$M_b($<двоичное число>$'0') = 2 \cdot M_b($<двоичное число>$);$
	
	$M_b($<двоичное число>$'1') = 2 \cdot M_b($<двоичное число>$) + 1.$
	
	$M_b$ - отображение синтаксических конструкций (\textit{двоичных чисел}) в математические объекты (\textit{натуральные десятичные числ}а).
\end{slide}

\begin{slide}
	\textbf{Денотационная семантика программы} определяется в терминах изменений состояний идеального компьютера.
	
	Отличие от операционной - использование строгих математических функция, а не запрограммированных алгоритмов.
	
	Состояние программы $s$ - ${<i_1, v_1>, <i_2, v_2>, \ldots , <i_n, v_n>}$
	
	$VARMAP(i_k, s)$ - значение переменной $i_k$ в состоянии $s$.
\end{slide}

	\textbf{Декларативная семантика} является характеристикой языков логического программирования. Например, Пролог.

\begin{slide}
	\textbf{Формальное определение семантики в жизни}
	
	Формальное определение семантики становится общепринятой нормаой для языков.
	
	Влияние формальных определений семантики < Влияние формальных грамматик.
